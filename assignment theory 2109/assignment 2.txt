#include<iostream>
using namespace std;
void main()
{
	int n = 1;
	for (int r = 1; r <= 4;r++)
	{
		for (int s = 3; s >= r; s--)
			cout << " ";
			for (int d = 1; d <= 2 * r - 1; d++)
			{
				if (n == 9)
					cout  <<  n;
				else
				{
					cout <<  n;
					n++;
				}
			}
			cout << endl;
	}
}
